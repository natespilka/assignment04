#Load in Libraries
library(haven)
library(tidyverse)
library(ipumsr)
library(srvyr)

#Load in data
ddi <- read_ipums_ddi("data/atus_00002.xml")
data <- read_ipums_micro(ddi)
## Use of data from IPUMS ATUS is subject to conditions including that users
## should cite the data appropriately. Use command `ipums_conditions()` for more
## details.
#Organize/clean data

color1 <- "#cd5c5c"
color2 <- "#5c5ccd"
color3 <- "#5ccd5c"

atusSurvey <- data %>%
  as_survey_design(weights = WT06)

#Create new and improved variables
atusSurveySubSet <- atusSurvey %>%
  mutate(
    married = case_when(
      MARST == 1 | MARST == 2 ~ 1,
      MARST == 4 | MARST == 5 ~ 0,
      MARST == 6 ~ 2),
    edu = case_when(
      EDUC == 021 ~ 0,
      EDUC == 040 ~ 1),
    # ,
    #   EDUC == 041 ~ 2,
    #   EDUC == 042 ~ 3,
    #   EDUC == 043 ~ 4),
    employed = case_when(
      EMPSTAT == 1 | EMPSTAT == 2 ~ 1,
      EMPSTAT == 3 | EMPSTAT == 4 ~ 0),
    wbh = case_when(
      RACE == 100 & HISPAN == 100 ~ 0,
      RACE == 110 ~ 1,
      HISPAN != 100 & HISPAN < 251 ~ 2)
  )

#clean some data - will add more
atusSurveySubSet <- atusSurveySubSet %>%
  filter(AGE < 122)

data2 <- atusSurveySubSet$variables
#it looks like unemployment is higher than online values
unEmployed <- atusSurveySubSet %>%
  filter(!is.na(employed)) %>%
  group_by(YEAR, employed) %>%
  summarize(unEmpProp = survey_prop()) %>%
  filter(employed == 0) %>%
  ungroup()

#unemployment rate for 2007 and 2009 segmented by race and education
unEmployed0709 <- atusSurveySubSet %>%
  filter(YEAR == 2007 | YEAR == 2009, !is.na(employed),!is.na(edu),!is.na(wbh)) %>%
  group_by(YEAR, edu, wbh, employed) %>%
  summarize(
    total = survey_prop()) %>%
  filter(employed == 0) %>%
  ungroup()

#Making the subtraction easier by making date-proportion columns
unEmployed0709Wide <- unEmployed0709 %>%
  pivot_wider(
    names_from = "YEAR",
    values_from = c("total", "total_se")
  )

#finding percent change from 2007 to 2009
unEmployed0709WidePC <- unEmployed0709Wide %>%
  mutate(
    prcntChnge = (total_2009 - total_2007)
  )

#Changing the appropriate variables to factors (edu and wbh)
unEmployed0709WidePC %>%
  mutate(
    edu = factor(edu, labels = c("High School Diploma", "Bachelor's Degree")),
    wbh = factor(wbh, labels = c("White", "Black", "Hispanic"))) %>%
  group_by(wbh) %>%
  ggplot(mapping = aes(x = edu, y = prcntChnge, fill = factor(wbh))) +
  geom_col(position = "dodge")+
  geom_text(mapping = aes(label = scales::percent(round(prcntChnge, digits = 3))), 
            position = position_dodge(width = 0.9), vjust = -1) +    
  scale_y_continuous(expand = expand_scale(mult = c(0, 0.1)),
                     breaks = c(0.03, 0.06, 0.09),
                     labels = scales::percent_format(scale = 100)) +
  ggtitle("Percent Increase in Unemployment After 2008 Financial Crisis (2007-2009)") +
  xlab(NULL) +
  ylab(NULL) +
  labs(fill = NULL) +
  scale_fill_manual(values=c(color1, color2, color3)) +
  theme(panel.background = element_blank(), 
  panel.grid.major = element_line(color = "black", size = 0.1),
  panel.grid.major.x = element_blank(),
  axis.ticks = element_blank())

#Only considering people of legal marrying age (<18yos in ATUS are coded as "Never Married")
atusSurveySubSetMar <- atusSurveySubSet %>%
  filter(AGE > 18)

data2 <- atusSurveySubSetMar$variables

mrydOvrTme <- atusSurveySubSetMar %>%
  filter(!is.na(married)) %>%
  group_by(YEAR, married) %>%
  summarize(
    total = survey_prop()) %>%

  mutate(
    married = factor(married, labels = c("Divorced/Separated", "Married", "Naver Married"))
  )


mrydOvrTme %>%
  group_by(married, YEAR) %>%
  ggplot(aes(x = YEAR, y = total, fill = married)) +
  geom_area(position = "fill") +
  scale_x_continuous(expand = expand_scale(mult = c(0, 0)),
                     limits = c(2003, 2017),
                     breaks = c(2004:2016)) +  
  scale_y_continuous(expand = expand_scale(mult = c(0, 0)),
                     breaks = c(0.25, 0.5, 0.75, 1),
                     labels = scales::percent) +
  ggtitle("Marital Status (2004-2016): did the 2008 financial crisis have an impact?") +
  xlab(NULL) +
  ylab(NULL) +
  labs(fill = NULL) +
